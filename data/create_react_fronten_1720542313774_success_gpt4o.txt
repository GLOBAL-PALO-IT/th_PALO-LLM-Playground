import React, { useState, useEffect } from 'react';
import './Maze.css';

const Maze = ({ maze }) => {
  const [position, setPosition] = useState({ x: 0, y: 0 });

  const handleKeyDown = (e) => {
    let newX = position.x;
    let newY = position.y;

    switch (e.key) {
      case 'ArrowUp':
        newY = Math.max(0, position.y - 1);
        break;
      case 'ArrowDown':
        newY = Math.min(maze.length - 1, position.y + 1);
        break;
      case 'ArrowLeft':
        newX = Math.max(0, position.x - 1);
        break;
      case 'ArrowRight':
        newX = Math.min(maze[0].length - 1, position.x + 1);
        break;
      default:
        break;
    }

    if (maze[newY][newX] !== 1) {
      setPosition({ x: newX, y: newY });
    }
  };

  useEffect(() => {
    window.addEventListener('keydown', handleKeyDown);
    return () => {
      window.removeEventListener('keydown', handleKeyDown);
    };
  }, [position]);

  return (
    <div className="maze">
      {maze.map((row, rowIndex) => (
        <div key={rowIndex} className="row">
          {row.map((cell, cellIndex) => (
            <div
              key={cellIndex}
              className={`cell ${cell === 1 ? 'wall' : ''} ${position.x === cellIndex && position.y === rowIndex ? 'player' : ''}`}
            ></div>
          ))}
        </div>
      ))}
    </div>
  );
};

export default Maze;
